Exemple de Stratégie Python

from quantflow import Strategy, Trade

class MomentumStrategy(Strategy):
    def __init__(self):
        self.lookback = 20
        self.threshold = 0.02
    
    def on_bar(self, bar):
        momentum = self.calculate_momentum(
            self.lookback
        )
        
        if momentum > self.threshold:
            self.buy(size=100)
        elif momentum < -self.threshold:
            self.sell(size=100)
